void _showMultiSelector(BuildContext context, StateSetter setState,
    List<String> items, List<String> selectedItem) {
  showDialog(
    context: context,
    builder: (BuildContext context) {
      return StatefulBuilder(
        builder: (BuildContext context, StateSetter setState) {
          return AlertDialog(
            backgroundColor: AppColors.white,
            content: Container(
              width: double.maxFinite,
              child: ListView.builder(
                shrinkWrap: true,
                itemCount: items.length,
                itemBuilder: (context, index) {
                  bool isSelected = selectedItem.contains(items[index]);
                  return InkWell(
                    onTap: () {
                      setState(() {
                        if (isSelected) {
                          selectedItem.remove(items[index]);
                        } else {
                          selectedItem.add(items[index]);
                        }
                      });
                    },
                    child: Padding(
                      padding: const EdgeInsets.symmetric(vertical: 8.0),
                      child: Row(
                        children: [
                          Icon(
                            isSelected
                                ? Icons.check_box
                                : Icons.check_box_outline_blank,
                            color: isSelected
                                ? AppColors.purple100
                                : AppColors.gainsBro,
                          ),
                          const SizedBox(width: 10),
                          TextWidget(
                            text: items[index],
                            textSize: 14,
                            fontWeight: FontWeight.w400,
                            color: AppColors.black80,
                          )
                        ],
                      ),
                    ),
                  );
                },
              ),
            ),
            // actions: [
            //   TextButton(
            //     onPressed: () {
            //       Navigator.of(context).pop();
            //     },
            //     child: const Text('Cancel'),
            //   ),
            //   TextButton(
            //     onPressed: () {
            //       Navigator.of(context).pop();
            //     },
            //     child: const Text('Done'),
            //   ),
            // ],
          );
        },
      );
    },
  );
}
